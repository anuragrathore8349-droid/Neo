### **Fix 14: Create Missing PriceAlerts Component**

**File to Create:** client/src/components/market/PriceAlerts.tsx

**Action 14.1:** Create component
```tsx
import React, { useState, useEffect } from 'react';
import { getPriceAlerts, createPriceAlert, deletePriceAlert } from '../../services/market.service';

export function PriceAlerts() {
  const [alerts, setAlerts] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadAlerts();
  }, []);

  const loadAlerts = async () => {
    try {
      const data = await getPriceAlerts();
      setAlerts(data);
    } catch (error) {
      console.error('Failed to load alerts', error);
    } finally {
      setLoading(false);
    }
  };

  const handleDelete = async (alertId: string) => {
    await deletePriceAlert(alertId);
    setAlerts(alerts.filter(a => a.id !== alertId));
  };

  return (
    <div className="price-alerts">
      <h2>Price Alerts</h2>
      {loading ? <p>Loading...</p> : (
        <div className="alerts-list">
          {alerts.map(alert => (
            <div key={alert.id} className="alert-item">
              <p>{alert.symbol} - {alert.targetPrice}</p>
              <button onClick={() => handleDelete(alert.id)}>Delete</button>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

  try {
    const alert = new MarketAlert({
      userId: req.user.id,
      ...req.body
    });
    await alert.save();
    res.status(201).json(alert);
  } catch (error) {
    res.status(400).json({ error: error.message });
  }

async deletePriceAlert(req, res) {
